# 路由与导航设计 — @kit.ArkUI router

> ⚠ API 20：导入使用 `@kit.ArkUI`，支持命名路由传参

## 导入方式

```typescript
// API 20 推荐
import { router } from '@kit.ArkUI';

// 带参数跳转
router.pushUrl({
  url: 'pages/HomePage',
  params: { userId: '...' }
});

// 返回
router.back();
```

## 页面层级结构

```
MainLayout (底部Tab容器)
├── HomePage (Tab1: 首页)
│   └── CourseDetailPage (详情页)
│       ├── [Tab页内 / 独立页面] Q1_KnowledgeGraphPage
│       ├── [Tab页内 / 独立页面] Q2_ControversyPage
│       └── [Tab页内 / 独立页面] Q3_AssessmentPage
│           └── AssessmentDetailPage (答题详情页 - 自由答题模式)
├── DiscussionHub (Tab2: 讨论区)
└── PersonalCenter (Tab3: 个人中心)
    ├── ArchiveManagementPage (归档管理)
    └── SettingsPage (设置)
```

## 路由表

```typescript
// AppRouter.ts
import { router } from '@kit.ArkUI';

export const ROUTES = {
  HOME: 'pages/HomePage',
  COURSE_DETAIL: 'pages/CourseDetailPage',
  ASSESSMENT_DETAIL: 'pages/AssessmentDetailPage',
  ARCHIVE: 'pages/ArchiveManagementPage',
  SETTINGS: 'pages/SettingsPage',
  DISCUSSION: 'pages/DiscussionHub',
  Q1_KNOWLEDGE_GRAPH: 'pages/Q1_KnowledgeGraphPage',
  Q2_CONTROVERSY: 'pages/Q2_ControversyPage',
  Q3_ASSESSMENT: 'pages/Q3_AssessmentPage',
} as const;

export type RoutePath = typeof ROUTES[keyof typeof ROUTES];

// 导航工具函数
export class AppRouter {
  static toCourseDetail(courseId: string): void {
    router.pushUrl({
      url: ROUTES.COURSE_DETAIL,
      params: { courseId }
    });
  }

  static toAssessmentDetail(courseId: string): void {
    router.pushUrl({
      url: ROUTES.ASSESSMENT_DETAIL,
      params: { courseId }
    });
  }

  static toArchive(): void {
    router.pushUrl({ url: ROUTES.ARCHIVE });
  }

  static toSettings(): void {
    router.pushUrl({ url: ROUTES.SETTINGS });
  }

  static back(): void {
    router.back();
  }
}
```

## 获取页面参数

```typescript
// 在目标页面中
import { router } from '@kit.ArkUI';

@Entry
@Component
struct CourseDetailPage {
  @State courseId: string = router.getParams()?.['courseId'] as string || '';

  aboutToAppear(): void {
    if (this.courseId) {
      this.loadCourse();
    }
  }
}
```

## 底部Tab导航 (MainLayout)

```typescript
// MainLayout.ets — 使用 Tabs 实现底部导航
@Entry
@Component
struct MainLayout {
  @State currentTabIndex: number = 0;

  build() {
    Column() {
      // 页面内容区
      this.currentPage();
    }
  }

  @Builder
  currentPage(): void {
    switch (this.currentTabIndex) {
      case 0: HomePage(); break;        // 首页
      case 1: PersonalCenter(); break;  // 个人中心
      case 2: DiscussionHub(); break;   // 讨论区
    }
  }
}
```
